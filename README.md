# brofile
```bash
git clone https://github.com/cooper-sloan/brofile ~/brofile
ln ~/brofile/bash_profile ~/.bash_profile
ln ~/brofile/gitconfig ~/.gitconfig
cd ~/Downloads
curl -O https://github.com/neovim/neovim/releases/download/v0.5.1/nvim-macos.tar.gz
tar xzvf nvim-macos.tar.gz
mv nvim-macos.tar /Applications/
cd -
```
